// 参数说明
$("input").autocomplete({
	search: [], // 待查询的数组
	result: "", // 显示结果容器的选择器
	itemHtml: "<li>", // 查询后产生选项html结构
	activeClass: "active",
	maxItemCount: 0, // 限制选项最大数目，0为不限制
	onClick: function(val) {} // 单击选项后的回调，并将选择的值作为参数传入，this指向点击选项的DOM对象
});
